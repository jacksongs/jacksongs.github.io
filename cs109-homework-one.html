# Given a poll_id, this function returns the XML data as a text string. This is the primary 'raw' scraper.
def get_poll_xml(poll_id):
    xml = requests.get('http://charts.realclearpolitics.com/charts/%s.xml'%poll_id).content
    return xml

# This function extract polls information from an XML string, and converts it to a pandas DataFrame.
def rcp_poll_data(xml):
    dom = web.Element(xml)
    series = dom.by_tag('series')
    graphs = dom.by_tag('graph')
    dates = series[0]
    results = {}
    for g in graphs:
        name = g.attributes[u'title']
        results[name] = [np.nan if t.content is u'' else float(t.content) for t in g]
    result = pd.DataFrame(results)
    result['date'] = pd.to_datetime([d.content for d in dates])
    result = result.sort('date')
    return result

# This functions plot the poll data.
# It was written by Chris Beaumont with some modifications by me to get the dates to display correctly.

def poll_plot(poll_id):

    xml = get_poll_xml(poll_id)
    data = rcp_poll_data(xml)
    colors = plot_colors(xml)

    #remove characters like apostrophes
    data = data.rename(columns = {c: _strip(c) for c in data.columns})
    
    #normalize poll numbers so they add to 100%    
    norm = data[colors.keys()].sum(axis=1) / 100    
    for c in colors.keys():
        data[c] /= norm
        
    fig, ax = plt.subplots()
    for label, color in colors.items():
        ax.plot(data.date.astype(DT.datetime), data[label], color=color, label=label)   
    plt.legend(loc='best')
    plt.xlabel("Date")

# This function removes non-letter characters from a word.
def _strip(s):
    return re.sub(r'[\W_]+', '', s)

# Given an XML document like the link above, this returns a python dictionary that maps a graph title to a graph color.
def plot_colors(xml):
    dom = web.Element(xml)
    result = {}
    for graph in dom.by_tag('graph'):
        title = _strip(graph.attributes['title'])
        result[title] = graph.attributes['color']
    return result

    dom = web.Element(html)
    part = dom.by_id('mymap')
    alllinks = part.by_tag('a')
    links = [str(a.attributes.get('href')) 
            for a in alllinks if 
            '2010/governor' in str(a.attributes.get('href')) and 
            'vs' in str(a.attributes.get('href'))]
    linkset = list(set(links))
    return linkset

# Now, for each governor race, we want to pull out the actual results in an easy-to-use format.

# This function scrapes the key fields in the table on each results page.
# It produces a dictionary with the results.
def race_result(url):
    content = requests.get(url).content
    dom = web.Element(content)
    cando = dom.by_id('candidates')    
    candies = len(cando)-1
    res = dom.by_tag('tr.final')
    total = float(res[0].children[3].content)+float(res[0].children[4].content)
    names = dom.by_tag('tr')
    results = {}
    c = 0
    while c < candies:
        results[names[3].children[c+3].content.split('(')[0][:-1]] = float(res[0].children[c+3].content)/total
        c+=1
    return results

# The next step is to start looking at the accuracy of the polls.

# This function determines the party from the colour.
def party_from_color(color):
    if color in ['#0000CC', '#3B5998']:
        return 'democrat'
    if color in ['#FF0000', '#D30015']:
        return 'republican'
    return 'other'

# This function builds a supercharged dataframe with:
#    candidate: Name of the candidate
#    forecast_length: Number of days before the election
#    percentage: The percent of poll votes a candidate has.
#    error: Difference between percentage and actual race result
#    party: Political party of the candidate
        
def error_data(url):
    id = id_from_url(url)
    xml = get_poll_xml(id)
    
    colors = plot_colors(xml)
    if len(colors) == 0:
        return pd.DataFrame()
    
    df = rcp_poll_data(xml)
    result = race_result(url)
    
    #remove non-letter characters from columns
    df = df.rename(columns={c: _strip(c) for c in df.columns})
    for k, v in result.items():
        result[_strip(k)] = v 
    
    candidates = [c for c in df.columns if c is not 'date']
        
    #turn into a timeseries...
    df.index = df.date
    
    #...so that we can resample at regular, daily intervals
    df = df.resample('D')
    df = df.dropna()
    
    #compute forecast length in days
    #(assuming that last forecast happens on the day of the election, for simplicity)
    forecast_length = (df.date.max() - df.date).values
    forecast_length = forecast_length / np.timedelta64(1, 'D')  # convert to number of days
    
    #compute forecast error
    errors = {}
    normalized = {}
    poll_lead = {}
    
    for c in candidates:
        #turn raw percentage into percentage of poll votes
        corr = df[c].values / df[candidates].sum(axis=1).values * 100.
        err = corr - result[_strip(c)] * 100
        
        normalized[c] = corr
        errors[c] = err
        
    n = forecast_length.size
    
    result = {}
    result['percentage'] = np.hstack(normalized[c] for c in candidates)
    result['error'] = np.hstack(errors[c] for c in candidates)
    result['candidate'] = np.hstack(np.repeat(c, n) for c in candidates)
    result['party'] = np.hstack(np.repeat(party_from_color(colors[_strip(c)]), n) for c in candidates)
    result['forecast_length'] = np.hstack(forecast_length for _ in candidates)
    
    result = pd.DataFrame(result)
    return result
