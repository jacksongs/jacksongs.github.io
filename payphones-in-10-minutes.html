# Now let's put it into a dataframe.
import pandas as pd
import numpy as np

# Here the longlist represents a list of rows, while short is a single row.
longlist = []
cats = [u'SECTOR', u'SSC_NAME', u'FEDERAL_ELECTORATE', u'MI_PRINX', u'NEXT_DISTANCE', u'STATE_ELECTORATE', u'distanceToFeature', u'STYLE', u'LONGITUDE', u'PHONE_ATTRIBUTES', u'STATE', u'POSTCODE', u'ADDRESS', u'LATITUDE', u'CABINET_ID', u'LGA_NAME', u'NEXT_PAYPHONE']
for f in feats:
    short = []
    for c in cats:
        try:
            short.append(f[u'properties'][c])
        except:
            short.append(np.nan)
    longlist.append(short)
df = pd.DataFrame(longlist)
df.columns = cats
